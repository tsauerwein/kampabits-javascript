Conditions

if-statement


var age = 34;

if (age > 18) {
	alert("You are older than 18!");
}

if the condition (between the parentheses) is true,
then the block between the braces "{" and "}" is executed.

if-else


if (age > 18) {
	alert("You are older than 18!");
} else {
	alert("You are younger than 18!");
}

013-logic-equals.htm

AND, OR and NOT

and - &&


if (a > 60 && a < 80) {
	// between 60 and 80
}

or - ||


if (b < 23 || b > 56) {
	// smaller than 23 or greater than 56
}

not - !


if (!(a > 23)) {
	// not greater than 23
}
