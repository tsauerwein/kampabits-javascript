	if (number < numberToGuess) {
		alert("You entered " + number + ". " 
			+ "The number we are looking for is bigger, try again!");

Introducing functions

Do you remember our program that identified the largest number in an array?


var numbers = [23, 89, 1, 23, 53, 103, 23];
var largest = 0;

for(var i = 0; i < numbers.length; i++) {
	if (numbers[i] > largest) {
		largest =  numbers[i];
	}
}
				
print(largest);

(we are assuming that the array is not empty and only contains positive numbers)

 

What if we would like to find the largest number in other arrays?

Introducing functions

A function to find the largest element in an array


var maximum = function(numbers) {
	var largest = 0;

	for(var i = 0; i < numbers.length; i++) {
		if (numbers[i] > largest) {
			largest =  numbers[i];
		}
	}

	return largest;
};
					

var numbers1 = [34, 23, 65, 12, 97];
print(maximum(numbers1));	// 97

var numbers2 = [121, 34, 12, 75, 31, 12];
print(maximum(numbers2));	// 121

(again we are assuming that the array is not empty and only contains positive numbers)

Constructors

Constructors - Functions to create new objects of a class


var Student = function(firstName, lastName, age) {
	this.firstName = firstName;
	this.lastName = lastName;
	this.age = age;

	this.printName = function() {
		print(this.firstName + " " + this.lastName);
	};
};
Note: The name of the constructor starts with a capital letter! 

var peter = new Student("Peter", "Anderson", 34);
peter.printName();	// Peter Anderson
var john = new Student("John", "Elison", 28);
new creates a new object using the Student constructor.
